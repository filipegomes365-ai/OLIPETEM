<script>
// Carrinho Simulado
let carrinho = [];
function adicionarCarrinho(nome, preco){
  carrinho.push({nome, preco});
  atualizarCarrinho();
}

function atualizarCarrinho(){
  let itens = document.getElementById('itens');
  let total = 0;
  itens.innerHTML = '';
  carrinho.forEach(item=>{
    itens.innerHTML += `<div class="carrinho-item">${item.nome} R$${item.preco}</div>`;
    total += item.preco;
  });
  document.getElementById('total').innerText = total;
  let msg = carrinho.map(i=>`${i.nome} R$${i.preco}`).join(', ');
  document.getElementById('whatsappCarrinho').href = `https://wa.me/5516999921350?text=Meu%20carrinho:%20${encodeURIComponent(msg)}%20Total:%20R$${total}`;
}
